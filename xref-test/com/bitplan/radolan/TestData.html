53    @Test
54    public void testPrecision() {
55      int[] precisions = { 2, 1, 0, -1, -2 };
56      float[] factors = { 100f, 10f, 1f, 0.1f, 0.01f };
57      Composite c = new Composite();
58      int i = 0;
59      for (int precision : precisions) {
60        c.setPrecision(precision);
61        assertEquals(factors[i], c.getPrecisionFactor(), 0.00001);
62        assertEquals(Math.pow(10, precision), c.getPrecisionFactor(), 0.00001);
63        i++;
64      }
65  
66    }

82    @Test
83    public void testSF1805301650() throws Throwable {
84      String history = "src/test/data/history/";
85      debug=true;
86      if (debug)
87        Composite.activateDebug();
88      File sfHistoryFile = new File(
89          history + "raa01-sf_10000-1805301650-dwd---bin.gz");
90      assertTrue(sfHistoryFile.getName(), sfHistoryFile.exists());
91      String url = sfHistoryFile.toURI().toURL().toExternalForm();
92      composite = new Composite(url);
93      checkLittleEndian(composite, "SF");
94      System.out.println(composite.getStatistics());
95      // show the original
96      super.testRadolan(url, 5, "sf-history2018-05-30_1650.png", null);
97  
98      // now get the content of the zip file
99      File zippedCsvFile = new File(
100         history + "raa01-sf_10000-1805301650-dwd---bin.txt.gz");
101     assertTrue(zippedCsvFile.getName(), zippedCsvFile.exists());
102     csv = gzipFileToString(zippedCsvFile);
103     assertNotNull(csv);
104     assertEquals(10275223, csv.length());
105     // apply it with readCSV and show results
106     super.testRadolan(url, 5, "sf-history2018-05-30_1650JS.png", readCsv);
107     // read original again
108     Composite.setPostInit(readCsv);
109     Composite cr = new Composite(url);
110     int errCount = 0;
111     int limit = 2000;
112     // check the whole grid
113 
114     for (int y = 0; y < cr.getGridHeight(); y++) {
115       for (int x = 0; x < cr.getGridWidth(); x++) {
116         float cval = composite.getValue(x, y);
117         float crval = cr.getValue(x, y);
118         if (Math.abs(crval - cval) > 0.001) {
119           errCount++;
120           if (errCount < limit) {
121             float d10 = (crval-cval)*10;
122             System.out.println(
123                 String.format("%3d,%3d: %5.1f != %5.1f d*10=%5.1f", x, y, cval, crval,d10));
124           }
125         }
126       }
127     }
128     assertEquals(0, errCount);
129   }

131   /**
132    * set the values from a csv file
133    */
134   public Consumer<Composite> readCsv = composite -> {
135     System.out.println(composite.getStatistics());
136     composite.getStatistics().clear();
137     Scanner scanner = new Scanner(csv);
138     scanner.useDelimiter(";|\\n");
139 
140     while (scanner.hasNext()) {
141       int y = Integer.parseInt(scanner.next());
142       int x = Integer.parseInt(scanner.next());
143       float val = Float.NaN;
144       String floatStr = scanner.next();
145       if (!floatStr.equals("Nan"))
146         val = Float.parseFloat(floatStr);
147       composite.setValue(x, y, val);
148     }
149     scanner.close();
150     System.out.println(composite.getStatistics());
151   };

153   /**
154    * get the utf-8 String content of the given gzipped file
155    * 
156    * @param file
157    * @return the text read
158    * @throws Exception
159    */
160   public static String gzipFileToString(File file) throws Exception {
161     GZIPInputStream gzipIn = new GZIPInputStream(new FileInputStream(file));
162     String text = IOUtils.toString(gzipIn, "UTF-8");
163     return text;
164   }

166   /**
167    * check the given composite
168    * 
169    * @param c
170    *          - the composite to check
171    * @param product
172    */
173   public void checkLittleEndian(Composite c, String product) {
174     assertEquals(product, c.getProduct());
175     assertEquals(900, c.getDx());
176     assertEquals(900, c.getDy());
177     // assertEquals(-1,c.getPrecision());
178     // assertEquals(0.1,c.getPrecisionFactor(),0.0001);
179     assertEquals(c.getDx() * c.getDy() * 2, c.getDataLength());
180     assertEquals(c.getDataLength() + c.header.length(), c.bytes.length);
181     assertEquals(c.PlainData.length, c.getDy());
182     assertEquals(Encoding.littleEndian, c.identifyEncoding());
183   }
