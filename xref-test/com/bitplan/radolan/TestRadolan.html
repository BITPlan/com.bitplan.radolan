66    /**
67     * test the OpenData for the given product ry/rw for the past minutes taken
68     * into account the given delay in minutes
69     * 
70     * @param product
71     *          - the product ry/rw
72     * @param minutes
73     *          - how many minutes of the previous period to cover
74     * @param minDelay
75     *          - the delay of this product
76     * @param startMin
77     *          - the start Minute e.g. 50 for rw 5 for ry
78     * @param minRaster
79     *          - the raster of this product e.g. 60 for rw 5 for ry
80     * @throws Exception 
81     */
82    public void testOpenDataRecent(String product, int minutes, int minDelay,
83        int startMin, int startRest, int minRaster) throws Exception {
84      LocalDateTime starttime = LocalDateTime
85          .ofInstant(Instant.now(), ZoneOffset.UTC)
86          .truncatedTo(ChronoUnit.MINUTES);
87      // safety marging
88      starttime = starttime.minus(Duration.ofMinutes(minDelay));
89      int count=0;
90      while ((starttime.getMinute() % startMin) != startRest) {
91        starttime = starttime.minus(Duration.ofMinutes(1));
92        if (count++ >20*365*24*60) {
93          throw new Exception("'endless' loop detected in finding recent data start");
94        }
95      }
96      LocalDateTime datetime = starttime;
97      LocalDateTime endtime = datetime.minus(Duration.ofMinutes(minutes));
98      for (; datetime.isAfter(
99          endtime); datetime = datetime.minus(Duration.ofMinutes(minRaster))) {
100       String url = (String.format(
101           "https://opendata.dwd.de/weather/radar/radolan/%s/raa01-%s_10000-%02d%02d%02d%02d%02d-dwd---bin",
102           product, product, datetime.getYear() % 2000, datetime.getMonthValue(),
103           datetime.getDayOfMonth(), datetime.getHour(), datetime.getMinute()));
104       String picture = String.format("%s-%04d-%02d-%02d_%02d%02d.png", product,
105           datetime.getYear(), datetime.getMonthValue(),
106           datetime.getDayOfMonth(), datetime.getHour(), datetime.getMinute());
107       testRadolan(url, 4, picture, null);
108     }
109   }

119   @Test
120   public void testHistory() {
121     if (!isTravis()) {
122       LocalDate date = LocalDate.of(2018, 8, 13);
123       LocalDate end = LocalDate.of(2018, 8, 15); // actually one day before this
124                                                  // ...
125       String knownUrl = "ftp://ftp-cdc.dwd.de/pub/CDC/grids_germany/daily/radolan/";
126       do {
127         String url = String.format(
128             knownUrl + "recent/raa01-sf_10000-%02d%02d%02d1650-dwd---bin.gz",
129             date.getYear() % 2000, date.getMonthValue(), date.getDayOfMonth());
130         if (!Composite.cacheForUrl(url, knownUrl).exists() || date.isEqual(end))
131           break;
132         testRadolan(url, 2, String.format("sf-%04d-%02d-%02d_1650.png",
133             date.getYear(), date.getMonthValue(), date.getDayOfMonth()), null);
134         date = date.plus(Period.ofDays(1));
135       } while (true);
136     }

159   @Test
160   public void testColors() {
161     FloatFunction<Color> heatmap = Vis.RangeMap(Vis.DWD_Style_Colors);
162     // taken from
163     // https://www.dwd.de/DE/leistungen/radolan/radolan_info/sf_karte.png?view=nasImage&nn=16102
164 
165     for (ColorRange colorRange : Vis.DWD_Style_Colors) {
166       // check upper and lower bound of color and a value in the middle
167       // all three should have the same color
168       float testValues[] = { colorRange.getFromValue(), colorRange.getToValue(),
169           (colorRange.getFromValue() + colorRange.getToValue()) / 2.0f };
170       for (float value : testValues) {
171         Color rColor = heatmap.apply(value);
172         assertEquals(colorRange.getColor(), rColor);
173       }
174     }
175   }
