82    static class MapView {
83      ImageView imageView;
84      Image image;
85      Pane drawPane;
86      private StackPane stackPane;

88      /**
89       * construct me from the given url
90       * 
91       * @param url
92       */
93      public MapView(String url) {
94        if (url != null)
95          getImageView(url);
96        drawPane = new Pane();
97        drawPane.setStyle(
98            "-fx-background-color: rgba(240, 240, 240, 0.05); -fx-background-radius: 10;");
99  
100       /*
101        * borderPane =new Pane(); borderPane.setStyle(
102        * "-fx-background-color: rgba(255, 128, 0, 0.05); -fx-background-radius: 10;"
103        * );
104        */
105       stackPane = new StackPane();
106       StackPane.setAlignment(imageView, Pos.CENTER);
107       stackPane.getChildren().addAll(imageView, drawPane);
108     }
109 
110     public Pane getPane() {
111       return stackPane;
112     }

114     /**
115      * get my ImageView
116      * 
117      * @param url
118      * @return the imageView
119      */
120     public ImageView getImageView(String url) {
121       // load the image
122       if (url != null)
123         image = new Image(url);
124 
125       // simple displays ImageView the image as is
126       imageView = new ImageView();
127       imageView.setImage(image);
128       imageView.setSmooth(true);
129       imageView.setCache(true);
130       return imageView;
131     }
132 
133     /**
134      * add a size Listener
135      */
136     public void addSizeListener(ObservableValue<Number> widthProperty,
137         ObservableValue<Number> heightProperty) {
138       ChangeListener<Number> widthListener = (observable, oldValue,
139           newValue) -> {
140         imageView.setFitWidth((double) newValue);
141         if (debug)
142           LOGGER.log(Level.INFO,
143               String.format("width %.0f->%.0f", oldValue, newValue));
144       };
145       ChangeListener<Number> heightListener = (observable, oldValue,
146           newValue) -> {
147         imageView.setFitHeight((double) newValue);
148         if (debug)
149           LOGGER.log(Level.INFO,
150               String.format("height %.0f->%.0f", oldValue, newValue));
151       };
152       widthProperty.addListener(widthListener);
153       heightProperty.addListener(heightListener);
154     }
155 
156     /**
157      * get the writeAbleImage (if available)
158      * 
159      * @return the writeAbleImage or null if the image is not writeAble
160      */
161     public WritableImage getWriteableImage() {
162       if (image instanceof WritableImage) {
163         return (WritableImage) image;
164       } else {
165         LOGGER.log(Level.INFO,
166             "image is not writeable will create a writeable copy");
167         WritableImage copyImage = copyImage(image);
168         image = copyImage;
169         imageView.setImage(image);
170         return copyImage;
171       }
172     }
173 
174     /**
175      * copy the given image to a writeable image
176      * 
177      * @param image
178      * @return a writeable image
179      */
180     public static WritableImage copyImage(Image image) {
181       int height = (int) image.getHeight();
182       int width = (int) image.getWidth();
183       PixelReader pixelReader = image.getPixelReader();
184       WritableImage writableImage = new WritableImage(width, height);
185       PixelWriter pixelWriter = writableImage.getPixelWriter();
186 
187       for (int y = 0; y < height; y++) {
188         for (int x = 0; x < width; x++) {
189           Color color = pixelReader.getColor(x, y);
190           pixelWriter.setColor(x, y, color);
191         }
192       }
193       return writableImage;
194     }
195 
196     /**
197      * add a size listener for the given stage
198      * 
199      * @param stage
200      */
201     public void addSizeListener(Stage stage) {
202       this.addSizeListener(stage.widthProperty(), stage.heightProperty());
203     }

206   public class BorderDraw {
207 
208     private MapView mapView;
209     private Borders borders;
210     private Color borderColor;
211     private Projection projection;
212 
213     /**
214      * construct me
215      * 
216      * @param mapView
217      * @param borderName
218      */
219     public BorderDraw(MapView mapView, Projection projection, String borderName,
220         Color borderColor) {
221       this.mapView = mapView;
222       this.projection = projection;
223       borders = new Borders(borderName);
224       this.borderColor = borderColor;
225     }
226 
227     /**
228      * draw the Borders
229      */
230     public void drawBorders() {
231       WritableImage image = mapView.getWriteableImage();
232       if (image == null) {
233         LOGGER.log(Level.WARNING, "can't draw Borders - image is null");
234         return;
235       }
236       IPoint prevIp = null;
237       List<DPoint> points = borders.getPoints();
238       LOGGER.log(Level.INFO, String.format("drawing %d border points",points.size()));
239       for (DPoint latlon : points) {
240         DPoint p = Translate.translate(projection, latlon.x, latlon.y);
241         IPoint ip = new IPoint(p);
242         // getScreenPointForLatLon(displayContext,borderPane,point);
243         double dist = ip.dist(prevIp);
244         if ((ip.x > 0 && ip.y > 0) && (dist < 40)) {
245            Line line = new Line(prevIp.x, prevIp.y, ip.x, ip.y);
246            line.setStrokeWidth(1); line.setStroke(borderColor);
247            mapView.drawPane.getChildren().add(line);
248           // image.getPixelWriter().setColor(ip.x, ip.y, borderColor);
249         }
250         prevIp = ip;
251       }
252       LOGGER.log(Level.INFO,"drawing done");
253     }
254    }

256   @Test
257   public void testDrawingBorders() throws Exception {
258     Borders.debug=true;
259     File imageFile = new File("src/test/data/image/rad_brd900x900.jpg");
260     for (String name : names) {
261       SampleApp.toolkitInit();
262       MapView mapView = new MapView(imageFile.toURI().toURL().toExternalForm());
263       Projection projection = new ProjectionImpl(900, 900);
264       Translate.calibrateProjection(projection);
265       BorderDraw borderDraw = new BorderDraw(mapView, projection, name,
266           Color.ORANGE);
267       SampleApp sampleApp = new SampleApp("BorderPlot", mapView.getPane());
268       sampleApp.show();
269       sampleApp.waitOpen();
270       double iwidth = mapView.image.getWidth();
271       double iheight = mapView.image.getHeight();
272       sampleApp.getStage().setWidth(mapView.image.getWidth());
273       sampleApp.getStage().setHeight(mapView.image.getHeight());
274       double width = sampleApp.getStage().getWidth();
275       double height = sampleApp.getStage().getHeight();
276       LOGGER.log(Level.INFO,
277           String.format("stage: %.0f x %.0f image: %.0f x %.0f ", width, height,
278               iwidth, iheight));
279       sampleApp.getStage().setHeight(mapView.image.getHeight() + 61);
280       mapView.addSizeListener(sampleApp.getStage());
281       Platform.runLater(() -> borderDraw.drawBorders());
282       Thread.sleep(SHOW_TIME);
283       sampleApp.close();
284       // Platform.exit();
285     }
