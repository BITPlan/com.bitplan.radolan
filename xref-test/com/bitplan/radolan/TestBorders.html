62    @Test
63    public void testBorders() throws Exception {
64      debug=true;
65      for (String name : names) {
66        Borders borders = new Borders(name);
67        if (debug)
68          LOGGER.log(Level.INFO, String.format("border %s has %d points", name,
69              borders.getPoints().size()));
70      }
71    }

73    static int SHOW_TIME = 4* 1000; // millisecs
74    private void setupStageLocation(Stage stage, int screenNumber) {
75      ObservableList<Screen> screens = Screen.getScreens();
76      Screen screen = screens.size() <= screenNumber ? Screen.getPrimary() : screens.get(screenNumber);
77  
78      Rectangle2D bounds = screen.getBounds();
79      boolean primary = screen.equals(Screen.getPrimary());    // WORKAROUND: this doesn't work nice in combination with full screen, so this hack is used to prevent going fullscreen when screen is not primary
80  
81      if(primary) {
82        stage.setX(bounds.getMinX());
83        stage.setY(bounds.getMinY());
84        stage.setWidth(bounds.getWidth());
85        stage.setHeight(bounds.getHeight());
86        stage.setFullScreen(true);
87      }
88      else {
89        stage.setX(bounds.getMinX());
90        stage.setY(bounds.getMinY());
91        stage.setWidth(bounds.getWidth());
92        stage.setHeight(bounds.getHeight());
93        stage.toFront();
94      }
95    }
96    @Test
97    public void testDrawingBorders() throws Exception {
98      Borders.debug=true;
99      File imageFile = new File("src/test/data/image/empty900x900.png");
100     for (String name : names) {
101       SampleApp.toolkitInit();
102       String url=imageFile.toURI().toURL().toExternalForm();
103       MapView mapView = new MapView(url);
104       Projection projection = new ProjectionImpl(900, 900);
105       Translate.calibrateProjection(projection);
106       BorderDraw borderDraw = new BorderDraw(mapView, projection, name,
107           Color.ORANGE);
108       SampleApp sampleApp = new SampleApp("BorderPlot", mapView.getPane());
109       sampleApp.show();
110       sampleApp.waitOpen();
111       // display on another monitor
112       // Platform.runLater(() ->this.setupStageLocation(sampleApp.getStage(), 2));
113       double iwidth = mapView.getImage().getWidth();
114       double iheight = mapView.getImage().getHeight();
115       sampleApp.getStage().setWidth(mapView.getImage().getWidth());
116       sampleApp.getStage().setHeight(mapView.getImage().getHeight());
117       double width = sampleApp.getStage().getWidth();
118       double height = sampleApp.getStage().getHeight();
119       LOGGER.log(Level.INFO,
120           String.format("stage: %.0f x %.0f image: %.0f x %.0f ", width, height,
121               iwidth, iheight));
122       sampleApp.getStage().setHeight(mapView.getImage().getHeight() + 61);
123       mapView.addSizeListener(sampleApp.getStage());
124       Platform.runLater(() -> borderDraw.drawBorders());
125       Thread.sleep(SHOW_TIME);
126       File snapShot=new File("/tmp/"+name.replace("/", "_")+".png");
127       Platform.runLater(() -> WaitableApp.saveAsPng(mapView.getPane(), snapShot));
128       sampleApp.close();
129       // Platform.exit();
130     }
