46    public boolean absequal(double a, double b, double epsilon) {
47      return Math.abs(a - b) < epsilon;
48    }

51    @Test
52    public void testResolution() {
53  
54      double srcLat = 48.173146;
55      double srcLon = 11.546604; // Munich
56      double dstLat = 53.534366;
57      double dstLon = 08.576135; // Bremerhaven
58      double expDist = 663.629945199998; // km
59  
60      Composite[] dummys = { Composite.NewDummy("SF", 900, 900),
61          Composite.NewDummy("SF", 450, 450), Composite.NewDummy("SF", 225, 225),
62          Composite.NewDummy("WX", 900, 1100), Composite.NewDummy("WX", 450, 550),
63          Composite.NewDummy("EX", 1400, 1500),
64          Composite.NewDummy("EX", 700, 750), };
65  
66      for (Composite comp : dummys) {
67        DPoint src = comp.translateLatLonToGrid(srcLat, srcLon);
68        DPoint dst = comp.translateLatLonToGrid(dstLat, dstLon);
69  
70        double resDist = Translate.dist(src.x * comp.getResX(), src.y * comp.getResY(),
71            dst.x * comp.getResX(), dst.y * comp.getResY());
72  
73        if (!absequal(resDist, expDist, 0.000001)) { // inaccuracy by 1mm
74          Errorf(
75              "dummy.Rx = %.2f, dummy.Ry = %.2f; distance: %.3f km expected: %.3f km)",
76              comp.getResX(), comp.getResY(), resDist, expDist);
77        }
78      }
79    }

113         { 43.9336, 02.3419, 0, 1500 }, { 43.8736, 18.2536, 1400, 1500 }, };
114 
115     Composite dummyPG = Composite.NewDummy("PG", 460, 460);
116     Composite dummyFZ = Composite.NewDummy("FZ", 450, 450);
117     Composite dummyRX = Composite.NewDummy("RX", 900, 900);

130           test.comp.getResX(), test.comp.getResY());
131       Logf("dummy%s: offx = %f; offy = %f\n", test.comp.getProduct(),
132           test.comp.getOffSetX(), test.comp.getOffSetY());
133       //debug=true;
134       for (double[] edge : test.edge) {

139         DPoint e = new DPoint(edge[2], edge[3]);

146         }
147         // inverse Polarsterographic projection
148         DPoint latlon2 = test.comp.translateGridToLatLon(e);
149         if (Translate.haversine(latlon2.x, latlon2.y, latlon.x, latlon.y) > 0.1) {
150           Errorf(
151               "dummy%s.TranslateXYtoLatLon(%.2f, %.2f) = (%.2f, %.2f); expected: (%.2f, %.2f)",
152               test.comp.getProduct(), e.x,e.y, latlon2.x, latlon2.y, latlon.x, latlon.y);
153         }
154 
155       }
156     }

187   @Test
188   public void testFormulas() {
189     assertEquals(Translate.rad(180),Math.toRadians(180),0.0001);
190     // Data from Table on page 13 of 
191     // https://www.dwd.de/DE/leistungen/radolan/radolan_info/radolan_radvor_op_komposit_format_pdf.pdf?__blob=publicationFile&v=10
192     PolarStereoProjection[] pspTable = { 
193         new PolarStereoProjection(Translate.junctionEast, Translate.junctionNorth, 0, -3185.02),
194         // 1.3
195         new PolarStereoProjection(3.5889, 46.9526, -523.4622, -4658.645),
196         new PolarStereoProjection(14.6209, 47.0705, 376.5378, -4658.645),
197         new PolarStereoProjection(15.7208, 54.7405, 376.5378, -3758.645),
198         new PolarStereoProjection(2.0715, 54.5877, -523.4622, -3758.645),
199         // 3.2 Middle European Grid
200         new PolarStereoProjection(2.3419, 43.9336, -673.4656656, -5008.642536),
201         new PolarStereoProjection(18.2536, 43.8736, 726.5343344, -5008.642536),
202         new PolarStereoProjection(21.6989, 56.4505, 726.5343344, -3508.642536),
203         new PolarStereoProjection(-0.8654, 56.5423, -673.4656656, -3508.642536)
204         };
205     for (PolarStereoProjection psp : pspTable) {
206       DPoint p = Translate.polarStereoProjection(psp.north, psp.east);
207       assertEquals("x",psp.x, p.x, 0.007);
208       assertEquals("y",psp.y, -p.y, 0.005);
209       DPoint ne=Translate.inversePolarStereoProjection(psp.x, -psp.y);
210       assertEquals("north",psp.north,ne.x,0.005);
211       assertEquals("east",psp.east,ne.y,0.005);
212     }
213   }

239   /**
240    * test the GridMode for the given Composite dummies
241    * 
242    * @param mode
243    * @param dummys
244    * @throws IOException
245    */
246   public void testGrid(GridMode mode, Composite... dummys) throws Exception {
247     // t.Helper()
248 
249     double offx = 0.0;
250     double offy = 0.0;
251 
252     switch (mode) {
253     case bottom:
254       offx = 0.0;
255       offy = 1.0;
256       break;
257     case center:
258       offx = 0.5;
259       offy = 0.5;
260       break;
261     default:
262       Fatalf("unknown grid mode %s", mode.toString());
263     }
264 
265     for (Composite comp : dummys) {
266       String lname = String.format("src/test/data/lambda_%s_%dx%d.txt", mode,
267           comp.getDy(), comp.getDx());
268       String pname = String.format("src/test/data/phi_%s_%dx%d.txt", mode,
269           comp.getDy(), comp.getDx());
270 
271       String lbuf = FileUtils.readFileToString(new File(lname), "UTF-8");
272       String pbuf = FileUtils.readFileToString(new File(pname), "UTF-8");
273 
274       // fortran format F8.5 means read 8 bytes, the last 5 bytes are decimal
275       // places
276       final int length = 8;
277 
278       int l = 0;
279       int p = 0;
280 
281       // Beschreibung-E-Produkte-Raster.pdf Radolan-Cons Version 1.0:
282       // "Die Dateien beginnen mit dem Referenzwert des Datenelements in der
283       // linken unteren Ecke des
284       // Komposits, spaltenweise von Westen nach Osten und zeilenweise von Süden
285       // nach Norden."
286       //
287       // "Für die Dateien mit der Bezeichnung _bottom beziehen sich die
288       // Koordinaten
289       // jeweils auf die linke untere Ecke jedes Datenelements. Für Dateien mit
290       // der Bezeichnung
291       // _center auf den Zentralpunkt."
292 
293       for (int y = comp.getDy() - 1; y >= 0; y--) {
294         for (int x = 0; x < comp.getDx(); x++) {
295 
296           // newlines can occur in input files.
297           while ('\n' == lbuf.charAt(l)) {
298             l++;
299           }
300           while ('\n' == pbuf.charAt(p)) {
301             p++;
302           }
303           String lstring = lbuf.substring(l, l + length).trim();
304           double lamda = Double.parseDouble(lstring);
305           /*
306            * if err != nil {
307            * t.Fatalf("invalid grid coordinate at (%d, %d): %#v %s", x, y,
308            * lstring, lname) }
309            */
310           l += length;
311 
312           String pstring = pbuf.substring(p, p + length).trim();
313           double phi = Double.parseDouble(pstring);
314           /*
315            * if err != nil {
316            * t.Fatalf("invalid grid coordinate at (%d, %d): %#v %s", x, y,
317            * pstring, pname) }
318            */
319           p += length;
320 
321           DPoint t = comp.translateLatLonToGrid(phi, lamda);
322           DPoint e = new DPoint(x + offx, y + offy);
323 
324           if (Translate.dist(t.x, t.y, e.x, e.y) > 0.01) { // 10m
325             Errorf(
326                 "dummy%s.Translate(%.2f, %.2f) = (%.2f, %.2f); expected: (%.2f, %.2f)",
327                 comp.getProduct(), phi, lamda, t.x, t.y, e.x, e.y);
328           }
329         }
330       }
331 
332       String lTail = lbuf.substring(l, lbuf.length() - 1);
333       if (lTail.length() > 0) {
334         Fatalf("unprocessed data remaining in %s", lname);
335       }
336       String pTail = pbuf.substring(p, pbuf.length() - 1);
337       if (pTail.length() > 0) {
338         Fatalf("unprocessed data remaining in %s", pname);
339       }
340     }
341   }
