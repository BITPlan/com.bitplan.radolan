41  /**
42   * manage a list of united nations location codes
43   * @author wf
44   *
45   */
46  public class UnLocodeManager implements JsonAble {
47    // prepare a LOGGER
48    protected static Logger LOGGER = Logger.getLogger("com.bitplan.geo");
49    
50    public static boolean debug=false;
51    
52    public static final transient String RESOURCE_NAME="unlocode_de.json";
53    public List<UnLocode> unLocodes=new ArrayList<UnLocode>();
54    public transient Map<String,UnLocode> unLocodeByName=new TreeMap<String,UnLocode>();
55    public transient Map<Integer,UnLocode> unLocodesByGridNumber=new TreeMap<Integer,UnLocode>();
56    
57    private static UnLocodeManager instance;
58    private UnLocodeManager() {};
59    /**
60     * singleton access
61     * @return my instance
62     */
63    public static UnLocodeManager getInstance() {
64      if (instance==null) {
65        // https://stackoverflow.com/a/21337734/1497139
66        try {
67          String json=IOUtils.toString(UnLocodeManager.class.getClassLoader().getResource(RESOURCE_NAME), "UTF-8");
68          JsonManagerImpl<UnLocodeManager> m=new JsonManagerImpl<UnLocodeManager>(UnLocodeManager.class);
69          instance=m.fromJson(json);
70        } catch (IOException e) {
71          LOGGER.log(Level.WARNING, "could not get "+RESOURCE_NAME+" from classpath/jar");
72        }
73      }
74      return instance;
75    }
76    
77    /**
78     * reinit me
79     */
80    public void reinit() {
81      for (UnLocode code:this.unLocodes) {
82        if (code.coords!=null) {
83          unLocodeByName.put(code.name, code);
84        }
85      }
86    }
87    
88    public static UnLocodeManager getEmpty() {
89      return new UnLocodeManager();
90    }
91    
92    /**
93     * find locations close to the given latitude and longitude
94     * as a sorted map by distance in km
95     * @param lat
96     * @param lon
97     * @return the map of cities
98     */
99    public Map<Double,UnLocode> lookup(double lat, double lon, double maxDist) {
100     Map<Double,UnLocode> cities=new TreeMap<Double,UnLocode>();
101     long startTime = System.nanoTime();
102  
103     // https://stackoverflow.com/questions/1260572/fastest-way-to-find-the-locationzip-city-state-given-latitude-longitude
104     for (UnLocode city:this.unLocodes) {
105       if (city.coords!=null) {
106         double citylat = city.getLat();
107         double citylon = city.getLon();
108         double dist=Translate.haversine(lat, lon, citylat, citylon);
109         if (dist<=maxDist) {
110           cities.put(dist, city);
111         }
112       }
113     }
114     long endTime = System.nanoTime();
115     long duration = (endTime - startTime) / 100000;
116     if (debug)
117       LOGGER.log(Level.INFO,
118           String.format("city lookup took %d msecs and returned %d results",
119               duration, cities.size()));
120     return cities;  
121   }
122   
123   /**
124    * lookup the given location
125    * @param name
126    * @return the location
127    */
128   public UnLocode lookup(String name) {
129     UnLocode code=this.unLocodeByName.get(name);
130     return code;
131   }
