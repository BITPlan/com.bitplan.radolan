24  package com.bitplan.geo;
25  
26  import gov.nasa.worldwind.geom.Angle;
27  
28  /**
29   * double precision Point Helper class
30   * @author wf
31   *
32   */
33  public class DPoint {
34    public double x;
35    public double y;
36    /**
37     * construct me
38     * @param x
39     * @param y
40     */
41    public DPoint(double x, double y) {
42      this.x=x;
43      this.y=y;
44    }
45    
46    public Angle getLat() {
47      return Angle.fromDegrees(x);
48    }
49    
50    public Angle getLon() {
51      return Angle.fromDegrees(y);
52    }
53    
54    public String toDMSString() {
55      String lat=getLat().toDMSString();
56      String lon=getLon().toDMSString();
57      return lat+" "+lon;
58    }
59    
60    public String getLatDMS() {
61      String latDMS = String.format("%s %s", getLat().toFormattedDMSString().substring(0, 9),
62          getLat().degrees >= 0.0 ? "N" : "S");
63      return latDMS;
64    }
65  
66    public String getLonDMS() {
67      String lonDMS = String.format("%s %s", getLon().toFormattedDMSString().substring(0, 9),
68          getLon().degrees >= 0.0 ? "E" : "W");
69      return lonDMS;
70    }
71    
72    /**
73     * format lat/lon to Degree minutes 
74     * @return a short formatted display version
75     */
76    public String toFormattedDMSString() {
77      String lat=getLatDMS();
78      String lon=getLonDMS();
79      String dms=(lat+" "+lon);
80      dms=dms.replaceAll("\\s+"," ").trim();
81      return dms;
82    }
83  }
