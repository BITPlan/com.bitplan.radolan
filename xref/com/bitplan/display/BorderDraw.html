24  package com.bitplan.display;
25  
26  import java.util.List;
27  import java.util.logging.Level;
28  import java.util.logging.Logger;
29  
30  import com.bitplan.geo.Borders;
31  import com.bitplan.geo.DPoint;
32  import com.bitplan.geo.IPoint;
33  import com.bitplan.geo.Projection;
34  
35  import cs.fau.de.since.radolan.Translate;
36  import javafx.scene.image.WritableImage;
37  import javafx.scene.paint.Color;
38  import javafx.scene.shape.Line;
39  
40  /**
41   * Helper class to draw borders
42   * 
43   * @author wf
44   *
45   */
46  public class BorderDraw {
47    public static boolean debug = false;
48  
49    protected static Logger LOGGER = Logger.getLogger("com.bitplan.display");
50  
51    private MapView mapView;
52    private Borders borders;
53    private Color borderColor;
54    private Projection projection;
55  
56    /**
57     * construct me
58     * 
59     * @param mapView
60     * @param borderName
61     */
62    public BorderDraw(MapView mapView, Projection projection, String borderName,
63        Color borderColor) {
64      this.mapView = mapView;
65      this.projection = projection;
66      borders = new Borders(borderName);
67      this.borderColor = borderColor;
68    }
69  
70    /**
71     * draw the Borders
72     */
73    public void drawBorders() {
74      WritableImage image = mapView.getWriteableImage();
75      if (image == null) {
76        LOGGER.log(Level.WARNING, "can't draw Borders - image is null");
77        return;
78      }
79      IPoint prevIp = null;
80      List<DPoint> points = borders.getPoints();
81      if (debug)
82        LOGGER.log(Level.INFO,
83            String.format("drawing %d border points", points.size()));
84      for (DPoint latlon : points) {
85        DPoint p = Translate.translate(projection, latlon.x, latlon.y);
86        IPoint ip = new IPoint(p);
87        // getScreenPointForLatLon(displayContext,borderPane,point);
88        double dist = ip.dist(prevIp);
89        if ((ip.x > 0 && ip.y > 0) && (dist < 40)) {
90          Line line = new Line(prevIp.x, prevIp.y, ip.x, ip.y);
91          line.setStrokeWidth(1);
92          line.setStroke(borderColor);
93          mapView.drawPane.getChildren().add(line);
94          // image.getPixelWriter().setColor(ip.x, ip.y, borderColor);
95        }
96        prevIp = ip;
97      }
98      if (debug)
99        LOGGER.log(Level.INFO, "drawing done");
100   }
101 }
