49  public class Zoom {
50    // prepare a LOGGER
51    protected static Logger LOGGER = Logger.getLogger("com.bitplan.radolan");
52    public static boolean debug = false;
53    PopOver popOver;
54    ImageView zoomView;
55    WritableImage zoomImage;
56    VBox vbox;
57    Label infoLabel;
58    private DisplayContext displayContext;
59    private int zoomFactor;
60  
61    /**
62     * create a zoom for the given display Context
63     * 
64     * @param displayContext
65     */
66    public Zoom(DisplayContext displayContext, int zoomFactor) {
67      this.zoomFactor=zoomFactor;
68      this.displayContext = displayContext;
69      zoomView = new ImageView();
70      zoomImage = new WritableImage((int) displayContext.zoomKm * zoomFactor,
71          (int) displayContext.zoomKm * zoomFactor);
72      zoomView.setImage(zoomImage);
73      //zoomView.setFitWidth(zoomImage.getWidth());
74      //zoomView.setFitHeight(zoomImage.getHeight());
75      infoLabel = new Label("no info");
76      infoLabel.setTextFill(Color.BLUE);
77      infoLabel.setBackground(new Background(
78          new BackgroundFill(Color.WHITE, CornerRadii.EMPTY, Insets.EMPTY)));
79      vbox = new VBox(infoLabel,zoomView);
80      popOver = new PopOver(vbox);
81    }
82  
83    /**
84     * arm me for the given grid point
85     * 
86     * @param gp
87     *          - the grid point
88     * @param vp
89     *          - the view point
90     * @return the string to describe the grid point
91     */
92    public String arm(IPoint gp, DPoint vp) {
93      RadarImage composite = displayContext.composite;
94      copyZoomContent(displayContext, gp);
95      // get the precipitation value for this point
96      float value = composite.getValue(gp.x, gp.y);
97      // get the location of the point as lat/lon
98      DPoint latlon = composite.translateGridToLatLon(new DPoint(gp.x, gp.y));
99      if (debug)
100       LOGGER.log(Level.INFO, String.format(
101           "scene %.0f,%.0f -> grid %d,%d %dx%d -> latlon %.2f,%.2f -> value %.0f mm",
102           vp.x, vp.y, gp.x, gp.y, composite.getGridWidth(),
103           composite.getGridHeight(), latlon.x, latlon.y, value));
104     // find the closest cities:
105     Map<Double, UnLocode> closestCities = UnLocodeManager.getInstance()
106         .lookup(latlon.x, latlon.y, 20);
107     String cityInfo = "";
108     if (closestCities.size() > 0) {
109       Entry<Double, UnLocode> cityEntry = closestCities.entrySet().iterator()
110           .next();
111       cityInfo = String.format(" near %s (%.1f km)",
112           cityEntry.getValue().getName(), cityEntry.getKey());
113     }
114     String displayMsg = String.format("%.1f %s%s\n%s", value,
115         composite.getDataUnit(), cityInfo, latlon.toFormattedDMSString());
116     String msg = String.format("%.0f,%.0f -> %s", vp.x, vp.y, displayMsg);
117     if (debug)
118       LOGGER.log(Level.INFO, msg);
119     infoLabel.setText(displayMsg);
120     return displayMsg;
121   }
122 
123   /**
124    * copy the zoom content for the given grid point
125    * @param displayContext
126    * @param gp
127    */
128   private void copyZoomContent(DisplayContext displayContext, IPoint gp) {
129     int half = (int) (displayContext.zoomKm / 2);
130     for (int x = gp.x - half; x <= gp.x + half; x++) {
131       for (int y = gp.y - half; y <= gp.y + half; y++) {
132         if (y >= 0 && y < displayContext.composite.getGridHeight())
133           if (x >= 0 && x < displayContext.composite.getGridWidth()) {
134             Color color = displayContext.image.getPixelReader().getColor(x, y);
135             int tx0 = x - gp.x + half;
136             int ty0 = y - gp.y + half;
137 
138             for (int tx = tx0*zoomFactor; tx < tx0*zoomFactor + zoomFactor; tx++) {
139               for (int ty = ty0*zoomFactor; ty < ty0*zoomFactor + zoomFactor; ty++) {
140                 if (tx < zoomImage.getWidth() && ty < zoomImage.getHeight())
141                   zoomImage.getPixelWriter().setColor(tx, ty, color);
142               }
143             }
144           }
145       }
146     }
147   }

149   /**
150    * trigger this zoom
151    * 
152    * @param node
153    */
154   public void triggerOnMouseEntered(Node node) {
155     node.setOnMouseEntered(mouseEvent -> {
156       // Show PopOver when mouse enters node
157       popOver.show(node);
158     });
159 
160     node.setOnMouseExited(mouseEvent -> {
161       // Hide PopOver when mouse exits node
162       popOver.hide();
163     });
164   }
