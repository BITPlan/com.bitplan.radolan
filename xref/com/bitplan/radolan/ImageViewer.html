56  public class ImageViewer extends WaitableApp {
57    private Image image;
58    private String title = "Image Viewer";
59    private ImageView imageView;
60    private DisplayContext displayContext;
61    private Pane drawPane;
62    private Scene scene;
63  
64    /**
65     * construct me from a DisplayContext
66     * 
67     * @param displayContext
68     */
69    public ImageViewer(DisplayContext displayContext) {
70      this.displayContext = displayContext;
71      this.title = displayContext.title;
72      this.image = displayContext.image;
73    }
74  
75    public ImageViewer() {
76    }
77  
78    @Override
79    public void start(Stage stage) {
80      super.start(stage);
81      if (displayContext != null) {
82        image = displayContext.image;
83      }
84      // load the image
85      if (image == null)
86        image = new Image("https://www.dwd.de/DWD/wetter/radar/rad_brd_akt.jpg");
87  
88      // simple displays ImageView the image as is
89      imageView = new ImageView();
90      imageView.setImage(image);
91      imageView.setSmooth(true);
92      imageView.setCache(true);
93  
94      ChangeListener<Number> widthListener = (observable, oldValue, newValue) -> {
95        imageView.setFitWidth((double) newValue);
96        if (debug)
97          LOGGER.log(Level.INFO,
98              String.format("width %.0f->%.0f", oldValue, newValue));
99      };
100     ChangeListener<Number> heightListener = (observable, oldValue,
101         newValue) -> {
102       imageView.setFitHeight((double) newValue);
103       if (debug)
104         LOGGER.log(Level.INFO,
105             String.format("height %.0f->%.0f", oldValue, newValue));
106     };
107     stage.widthProperty().addListener(widthListener);
108     stage.heightProperty().addListener(heightListener);
109     // imageView.fitWidthProperty().bind(stage.widthProperty());
110     // imageView.fitHeightProperty().bind(stage.heightProperty());
111     // imageView.setPreserveRatio(true);
112 
113     drawPane = new Pane();
114     drawPane.setStyle(
115         "-fx-background-color: rgba(240, 240, 240, 0.05); -fx-background-radius: 10;");
116 
117     /*
118      * borderPane =new Pane(); borderPane.setStyle(
119      * "-fx-background-color: rgba(255, 128, 0, 0.05); -fx-background-radius: 10;"
120      * );
121      */
122     StackPane stackPane = new StackPane();
123     StackPane.setAlignment(imageView, Pos.CENTER);
124     stackPane.getChildren().addAll(imageView, drawPane);
125 
126     // container with a fill property
127     scene = new Scene(stackPane);
128     scene.setFill(Color.WHITE);
129 
130     // inform the display context
131     if (displayContext != null) {
132       displayContext.imageView = imageView;
133       displayContext.drawPane = drawPane;
134       stage.setWidth(displayContext.image.getWidth());
135       stage.setHeight(displayContext.image.getHeight());
136     }
137     stage.setTitle(title);
138     stage.setScene(scene);
139     stage.sizeToScene();
140     stage.show();
141     if (debug)
142       stackPane.setOnMouseClicked(event -> {
143         showSizes();
144       });
145     // JavaFX: stage's minHeight considering titlebar's height
146     // https://stackoverflow.com/a/43346746/1497139
147     // stage.setMinHeight(stage.getHeight());
148     // stage.setMinWidth(stage.getWidth());
149     // How to make javafx.scene.Scene resize while maintaining an aspect ratio?
150     // https://stackoverflow.com/a/18638505/1497139
151     // stage.minWidthProperty().bind(scene.heightProperty());
152     // stage.minHeightProperty().bind(scene.widthProperty());
153   }
154 
155   /**
156    * show the sizes of stage, scene, imageView and Image
157    */
158   public void showSizes() {
159     System.out.println("Sizes: ");
160     ObservableList<Screen> screens = Screen.getScreensForRectangle(stage.getX(),
161         stage.getY(), stage.getWidth(), stage.getHeight());
162     for (Screen screen : screens) {
163       Rectangle2D s = screen.getVisualBounds();
164       showSize(screen, s.getWidth(), s.getHeight());
165     }
166     showSize(stage, stage.getWidth(), stage.getHeight());
167     showSize(scene, scene.getWidth(), scene.getHeight());
168     showSize(imageView, imageView.getFitWidth(), imageView.getFitHeight());
169     // showSize(borderPane,borderPane.getWidth(),borderPane.getHeight());
170     showSize(drawPane, drawPane.getWidth(), drawPane.getHeight());
171     showSize(image, image.getHeight(), image.getWidth());
172   }
173 
174   public void showSize(Object o, double width, double height) {
175     System.out.println(String.format("%s %.0f x %.0f",
176         o.getClass().getSimpleName(), width, height));
177   }
