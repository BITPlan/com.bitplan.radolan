61    /**
62     * get the Image for the given composite
63     * https://www.dwd.de/DE/leistungen/radarniederschlag/rn_info/download_niederschlagsbestimmung.pdf?__blob=publicationFile&v=4
64     * 
65     * @param displayContext
66     *          - the container for the image and it's details
67     * @throws Exception
68     */
69    public static void getImage(DisplayContext displayContext) throws Exception {
70      // get the Image for the displayContext
71      Platform.runLater(() -> {
72        getImageContent(displayContext);
73        // draw borders and mesh if asked for
74        if (displayContext.composite.isProjection()) {
75          drawBorders(displayContext);
76          drawMesh(displayContext);
77        }
78      });
79    }

81    /**
82     * draw Borders
83     * 
84     * @param displayContext
85     *          - the image and it's details
86     */
87    protected static void drawBorders(DisplayContext displayContext) {
88      /*
89       * Pane borderPane = displayContext.borderPane; if (borderPane == null)
90       * return; Platform.runLater(() -> { borderPane.getChildren().clear(); if
91       * (debug) { drawCross(borderPane, 2, Color.rgb(0xff, 0x00, 0x00, 0.5)); }
92       * });
93       */
94      RadarImage comp = displayContext.composite;
95      WritableImage image = displayContext.getWriteableImage();
96      String msg = String.format("detected grid: %.1f km * %.1f km\n",
97          comp.getGridWidth() * comp.getResX(),
98          comp.getGridHeight() * comp.getResY());
99      if (debug)
100       LOGGER.log(Level.INFO, msg);
101     Borders borders = new Borders(displayContext.borderName);
102     IPoint prevIp = null;
103     for (DPoint latlon : borders.getPoints()) {
104       DPoint p = displayContext.composite.translateLatLonToGrid(latlon.x,
105           latlon.y);
106       IPoint ip = new IPoint(p);
107       // getScreenPointForLatLon(displayContext,borderPane,point);
108       double dist = ip.dist(prevIp);
109       if ((ip.x > 0 && ip.y > 0) && (dist < 30)) {
110         /*
111          * Line line = new Line(prevIp.x, prevIp.y, ip.x, ip.y);
112          * line.setStrokeWidth(2); line.setStroke(borderColor);
113          * Platform.runLater(() -> { borderPane.getChildren().add(line); });
114          */
115         image.getPixelWriter().setColor(ip.x, ip.y, borderColor);
116       }
117       prevIp = ip;
118     }
119   }

121   /**
122    * draw a coordinate mesh
123    * 
124    * @param displayContext
125    *          - the image and it's details
126    */
127   protected static void drawMesh(DisplayContext displayContext) {
128     RadarImage comp = displayContext.composite;
129     WritableImage image = displayContext.getWriteableImage();
130     // draw mesh
131     // loop over east and north
132     for (double e = 1.0; e < 16.0; e += 0.1) {
133       for (double n = 46.0; n < 55.0; n += 0.1) {
134         double edist = Math.abs(e - Math.round(e));
135         double ndist = Math.abs(n - Math.round(n));
136         if ((edist < 0.01) || (ndist < 0.01)) {
137           DPoint dp = comp.translateLatLonToGrid(n, e);
138           IPoint ip = new IPoint(dp);
139           if (ip.x >= 0 && ip.y >= 0 && ip.x < comp.getGridWidth()
140               && ip.y < comp.getGridHeight())
141             image.getPixelWriter().setColor(ip.x, ip.y, meshColor);
142         }
143       }
144     }
145   }

147   /**
148    * get the Image for the given composite and color map
149    * 
150    * @param displayContext
151    */
152   public static void getImageContent(DisplayContext displayContext) {
153     RadarImage c = displayContext.composite;
154     FloatFunction<Color> colorMap = displayContext.heatmap;
155     int width = c.getGridWidth();
156     int height = c.getGridHeight();
157     WritableImage img = displayContext.getWriteableImage();
158     if (img == null)
159       return;
160     PixelWriter pw = img.getPixelWriter();
161     for (int y = 0; y < height; y++) {
162       for (int x = 0; x < width; x++) {
163         float value = c.getValue(x, y);
164         Color color = colorMap.apply(value);
165         pw.setColor(x, y, color);
166       }
167     }
168   }

170   /**
171    * activate the onShow Event of the given tooltip
172    * 
173    * @param displayContext
174    *          - details of the image e.g. view and tooltip
175    */
176   public static void activateEvents(DisplayContext displayContext) {
177     ImageView imageView = displayContext.imageView;
178     RadarImage composite = displayContext.composite;
179     Pane drawOnGlass = displayContext.drawPane;
180     
181     Bounds viewBounds = imageView.getBoundsInParent();
182     String imsg = String.format(
183         "events activated in a view with size %.0f x %.0f for composite %d x %d",
184         viewBounds.getWidth(), viewBounds.getHeight(), composite.getGridWidth(),
185         composite.getGridHeight());
186     if (debug)
187       LOGGER.log(Level.INFO, imsg);
188 
189     // initial addLocation
190     Platform.runLater(() -> addLocation(displayContext));
191 
192     //  handle a mouseclick
193     displayContext.drawPane.setOnMouseClicked(event -> {
194       // view point
195       DPoint vp = new DPoint(event.getSceneX(), event.getSceneY());
196       // grid point
197       IPoint gp = composite.translateViewToGrid(vp, viewBounds.getWidth(),
198           viewBounds.getHeight());
199       Zoom zoom=new Zoom(displayContext,12);
200       String text=zoom.arm(gp,vp);
201       Circle circle = drawCircleWithText(displayContext.drawPane, text, 4, Color.BLUE, vp.x, vp.y);
202       zoom.popOver.show(circle);
203     });
204    
205     ChangeListener<Number> sizeListener = (observable, oldValue, newValue) -> {
206       // too slow
207       // drawBorders(displayContext);
208       addLocation(displayContext);
209     };
210     drawOnGlass.widthProperty().addListener(sizeListener);
211     drawOnGlass.heightProperty().addListener(sizeListener);
212   }

214   /**
215    * add a location if there is one
216    * 
217    * @param displayContext
218    */
219   public static void addLocation(DisplayContext displayContext) {
220     // make sure the necessary prerequisites are there
221     if (displayContext.location == null || displayContext.composite == null)
222       return;
223     Pane drawOnGlass = displayContext.drawPane;
224     // clear the drawPane
225     drawOnGlass.getChildren().clear();
226     UnLocode loc = displayContext.location;
227     DPoint latlon = new DPoint(loc.getLat(), loc.getLon());
228     DPoint gpd = displayContext.composite.translateLatLonToGrid(latlon.x,
229         latlon.y);
230     IPoint gp = new IPoint(gpd);
231     double value = displayContext.composite.getValue(gp.x, gp.y);
232     // Position now needs to be adapted to screen size
233     DPoint vp = displayContext.composite.translateGridToView(gp, drawOnGlass.getWidth(),
234         drawOnGlass.getHeight());
235     String text = String.format("%s - %.1f mm", loc.getName(), value);
236     Circle circle = drawCircleWithText(displayContext.drawPane, text, 4, Color.BLUE, vp.x, vp.y);
237     Zoom zoom=new Zoom(displayContext,12);
238     zoom.arm(gp, vp);
239     zoom.triggerOnMouseEntered(circle);
240   }

242   /**
243    * draw a circle with given text on the given pane
244    * 
245    * @param pane
246    * @param text
247    * @param radius
248    * @param color
249    * @param x
250    * @param y
251    */
252   public static Circle drawCircleWithText(Pane pane, String text, double radius,
253       Color color, double x, double y) {
254     Circle circle = new Circle();
255     circle.setRadius(radius);
256     circle.setFill(color);
257     circle.setTranslateX(x);
258     circle.setTranslateY(y);
259 
260     Label label = new Label(text);
261     label.setTranslateX(x + radius);
262     label.setTranslateY(y + radius);
263     label.setTextFill(color);
264     pane.getChildren().addAll(circle, label);
265     return circle;
266   }

268   /**
269    * draw a cross on the given pane with the given stroke width and color
270    * 
271    * @param pane
272    * @param strokeWidth
273    * @param color
274    */
275   public static void drawCross(Pane pane, double strokeWidth, Color color) {
276     double w = pane.getWidth();
277     double h = pane.getHeight();
278     Line line = new Line(0, 0, w, h);
279     line.setStrokeWidth(strokeWidth);
280     line.setStroke(color);
281     Line line2 = new Line(w, 0, 0, h);
282     line2.setStrokeWidth(strokeWidth);
283     line2.setStroke(color);
284     pane.getChildren().addAll(line, line2);
285   }
