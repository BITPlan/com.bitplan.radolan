94     * get the integer for the given color
95     * 
96     * @param c
97     * @return the color integer
98     */
99    public static int getIntFromColor(Color c) {
100     int R = (int) Math.round(255 * c.getRed());
101     int G = (int) Math.round(255 * c.getGreen());
102     int B = (int) Math.round(255 * c.getBlue());
103     int A = (int) Math.round(255 * c.getOpacity());
104     A = (A << 24) & 0xFF000000;
105     R = (R << 16) & 0x00FF0000;
106     G = (G << 8) & 0x0000FF00;
107     B = B & 0x000000FF;
108 
109     return A | R | G | B;
110   }

126   // Graymap returns a grayscale gradient function between min and max. A
127   // compression function is used to
128   // make logarithmic scales possible.
129   public static FloatFunction<Color> Graymap(float pMin, float pMax,
130       FloatFunction<Float> compression) {
131     final float min = compression.apply(pMin);
132     final float max = compression.apply(pMax);
133 
134     FloatFunction<Color> gradient = (val) -> {
135       val = compression.apply(val);
136 
137       if (val < min) {
138         return Color.BLACK; // black
139       }
140 
141       double p = (val - min) / (max - min);
142       if (p > 1) {
143         p = 1;
144       }
145 
146       byte l = (byte) (0xFF * p);
147       return Color.rgb(l, l, l, 0xFF);
148     };
149     return gradient;
150   }

152   // Radialmap returns a dichromatic gradient from min to 0 to max which can
153   // be used for doppler radar radial velocity products.
154   public static FloatFunction<Color> Radialmap(float min, float max,
155       FloatFunction<Float> compression) {
156     FloatFunction<Color> radialmap = (val) -> {
157       if (val != val) {
158         return Color.BLACK; // black
159       }
160 
161       float base = Math.max(Math.abs(min), Math.abs(max));
162       float p = compression.apply(Math.abs(val)) / compression.apply(base);
163 
164       if (p > 1) {
165         p = 1;
166       }
167       int lev = (byte) (0xFF * p);
168 
169       int non = 0x00;
170       if (Math.abs(val) <= 1) {
171         lev = 0xFF;
172         non = 0xCC;
173       }
174 
175       if (val < 0) {
176         return Color.rgb(non, lev, lev, 0xFF);
177       }
178 
179       return Color.rgb(lev, non, non, 0xFF);
180     };
181     return radialmap;
182   }

246   /**
247    * get a Color Function based on a list of colorRanges
248    * @param colorRanges
249    * @return the RangeMap
250    */
251   public static FloatFunction<Color> RangeMap(ColorRange[] colorRanges) {
252     FloatFunction<Color> dwdStyleMap = (val) -> { 
253       for (ColorRange colorRange:colorRanges) {
254         if (val>=colorRange.getFromValue() && val<=colorRange.getToValue()) {
255           return colorRange.getColor();
256         }
257       }
258       return Color.rgb(0xE8, 0xE8, 0xE8, 0.75); // gray 3/4 opaque
259     };
260     return dwdStyleMap;
261   }
262   /**
263    * Heatmap returns a color gradient between pMin and pMax. A compression
264    * function is used to make logarithmic scales possible.
265    * 
266    * @param pMin
267    * @param pMax
268    * @param compression
269    * @return the gradient heatmap
270    */
271   public static FloatFunction<Color> Heatmap(float pMin, float pMax,
272       FloatFunction<Float> compression) {
273     final float min = compression.apply(pMin);
274     final float max = compression.apply(pMax);
275 
276     FloatFunction<Color> heatMap = (val) -> {   
277       val = compression.apply(val);
278       float cmin=min;
279       float cmax=max;
280    
281       if (val < cmin) {
282         return Color.rgb(0xE8, 0xE8, 0xE8, 0.75); // gray 3/4 opaque
283       }
284       if (val > cmax) {
285         return Color.PURPLE;
286       }
287       if (Float.isNaN(val)) {
288         return COLOR_INVALID; // 95% opaque
289       }
290 
291       double p = (val - min) / (max - min);
292       if (p > 1) { // limit
293         p = 1;
294       }
295       int h = (int) (Math.round(360 - (330 * p) + 240) % 360);
296 
297       double s = 1.0; // saturation
298       double l = 0.5 * p + 0.25; // lightness
299 
300       // adapted from https://en.wikipedia.org/wiki/HSL_and_HSV#From_HSL
301       double c = (1 - Math.abs(2 * l - 1)) * s; // calculate chroma
302 
303       int hh = h / 60;
304       double x = c * (1 - Math.abs(hh % 2) - 1);
305 
306       double rr = 0;
307       double gg = 0;
308       double bb = 0;
309       switch (hh) {
310       case 0:
311         rr = c;
312         gg = x;
313         bb = 0;
314         break;
315       case 1:
316         rr = x;
317         gg = c;
318         bb = 0;
319         break;
320       case 2:
321         rr = 0;
322         gg = c;
323         bb = x;
324         break;
325       case 3:
326         rr = 0;
327         gg = x;
328         bb = c;
329         break;
330       case 4:
331         rr = x;
332         gg = 0;
333         bb = c;
334         break;
335       case 5:
336         rr = c;
337         gg = 0;
338         bb = x;
339         break;
340       }
341 
342       double m = l - c / 2;
343       int r = (int) (0xFF * (rr + m));
344       int g = (int) (0xFF * (gg + m));
345       int b = (int) (0xFF * (bb + m));
346 
347       if (r >= 0 && g >= 0 && b >= 0) {
348         return Color.rgb(r, g, b, 1);
349       } else {
350         return Color.ORANGE;
351       }
352     };
353     return heatMap;
354   }
