41    // Z-R relationship
42    static class ZR  {
43      // intermediate caching
44      double c1; // 10*b
45      double c2; // a^(-1/b)
46      double c3; // 10^(1/(10*b))
47      double c4; // 10 * log10(a)
48    }
49  
50    // Common Z-R relationships
51    
52    static ZR  Aniol80          = NewZR(256, 1.42); // operational use in germany, described in [6]
53    static ZR  Doelling98       = NewZR(316, 1.50); // operational use in switzerland
54    static ZR  JossWaldvogel70  = NewZR(300, 1.50);
55    static ZR   MarshallPalmer55 = NewZR(200, 1.60); // operational use in austria
56   
57    // New Z-R returns a Z-R relationship mathematically expressed as Z = a * R^b
58    public static ZR NewZR(double A, double B)  {
59      ZR zr=new ZR();
60      zr.c1 = 10.0 * B;
61      zr.c2 = Math.pow(A, -1.0/B);
62      zr.c3 = Math.pow(10.0, 1/zr.c1);
63      zr.c4 = 10.0 * Math.log10(A);
64  
65      return zr;
66    }
67  
68    // PrecipitationRate returns the estimated precipitation rate in mm/h for the given
69    // reflectivity factor and Z-R relationship.
70    public static double PrecipitationRate(ZR relation ,double dBZ) {
71      return relation.c2 * Math.pow(relation.c3, dBZ);
72    }
73  
74    // Reflectivity returns the estimated reflectivity factor for the given precipitation
75    // rate (mm/h) and Z-R relationship.
76    public static double Reflectivity(ZR relation, double rate)  {
77      return relation.c4 + relation.c1 * Math.log10(rate);
78    }
79  
80    // toDBZ converts the given radar video processor values (rvp-6) to radar reflectivity
81    // factors in decibel relative to Z (dBZ).
82    public static float toDBZ(float rvp)  {
83      return rvp/2.0f - 32.5f;
84    }
85  
86    // toRVP6 converts the given radar reflectivity factors (dBZ) to radar video processor
87    // values (rvp-6).
88    public static double toRVP6(float dBZ) {
89      return (dBZ + 32.5) * 2;
90    }
91  
92    // rvp6Raw converts the raw value to radar video processor values (rvp-6) by applying the
93    // products precision field.
94    public static double rvp6Raw(Composite c, int value)  {
95      double rvalue=value * c.getPrecisionFactor();
96      return rvalue;
97    }
