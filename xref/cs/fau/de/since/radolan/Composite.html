53  //is available at the Global Basic Dataset (http://www.dwd.de/DE/leistungen/gds/gds.html).
54  //The obtained results can be processed and visualized with additional functions.
55  //
56  //Currently the national grid [1][4] and the extended european grid [5] are supported.
57  //Tested input products are PG, FZ, SF, RW, RX and EX. Those can be considered working with
58  //sufficient accuracy.
59  //
60  //In cases, where the publicly available format specification is unprecise or contradictory,
61  //reverse engineering was used to obtain reasonable approaches.
62  //Used references:
63  //
64  //[1] https://www.dwd.de/DE/leistungen/radolan/radolan_info/radolan_radvor_op_komposit_format_pdf.pdf
65  //[2] https://www.dwd.de/DE/leistungen/gds/weiterfuehrende_informationen.zip
66  //[3]  - legend_radar_products_fz_forecast.pdf
67  //[4]  - legend_radar_products_pg_coordinates.pdf
68  //[5]  - legend_radar_products_radolan_rw_sf.pdf
69  //[6] https://www.dwd.de/DE/leistungen/radarniederschlag/rn_info/download_niederschlagsbestimmung.pdf
70  
71  //Radolan radar data is provided as single local sweeps or so called composite
72  //formats. Each composite is a combined image consisting of mulitiple radar
73  //sweeps spread over the composite area.
74  //The 2D composite c has a an internal resolution of c.Dx (horizontal) * c.Dy
75  //(vertical) records covering a real surface of c.Dx * c.Rx * c.Dy * c.Dy
76  //square kilometers.
77  //The pixel value at the position (x, y) is represented by
78  //c.Data[ y ][ x ] and is stored as raw float value (NaN if the no-data flag
79  //is set). Some 3D radar products feature multiple layers in which the voxel
80  //at position (x, y, z) is accessible by c.DataZ[ z ][ y ][ x ].
81  //
82  //The data value is used differently depending on the product type:
83  //(also consult the DataUnit field of the Composite)
84  //
85  //Product label            | values represent         | unit
86  //-------------------------+--------------------------+------------------------
87  //PG, PC, PX*, ...         | cloud reflectivity       | dBZ
88  //RX, WX, EX, FZ, FX, ...  | cloud reflectivity       | dBZ
89  //RW, SF,  ...             | aggregated precipitation | mm/interval
90  //PR*, ...                 | doppler radial velocity  | m/s
91  //
92  //The cloud reflectivity (in dBZ) can be converted to rainfall rate (in mm/h)
93  //via PrecipitationRate().
94  //
95  //The cloud reflectivity factor Z is stored in its logarithmic representation dBZ:
96  //dBZ = 10 * log(Z)
97  //Real world geographical coordinates (latitude, longitude) can be projected into the
98  //coordinate system of the composite by using the translation method:
99  //// if c.HasProjection
100 //x, y := c.Translate(52.51861, 13.40833) // Berlin (lat, lon)
101 //
102 //dbz := c.At(int(x), int(y))         // Raw value is Cloud reflectivity (dBZ)
103 //rat := radolan.PrecipitationRate(radolan.Doelling98, dbz) // Rainfall rate (mm/h) using Doelling98 as Z-R relationship
104 //
105 //fmt.Println("Rainfall in Berlin [mm/h]:", rat)
106 //

286   /**
287    * construct me from an url - read the data immediately from url or if the
288    * cache is active and the content is available from the cache
289    * 
290    * @param url
291    * @throws Throwable
292    */
293   public Composite(String url) throws Throwable {
294     this();
295     if (debug)
296       LOGGER.log(Level.INFO, "getting composite for url " + url);
297     this.url = checkCache(url);
298 
299     InputStream inputStream = new URL(url).openStream();
300     read(inputStream);
301     init();
302   }

304   /**
305    * if the cache is not active or the url is not starting with a known url then
306    * return the url as is when the local cache is active then check if the url
307    * content is not available locally and if needed then read the url content to
308    * the cache after having made sure the file is available locally return the
309    * url for the file
310    * 
311    * @param url
312    * @return the (potentially replaced)
313    * @throws Exception
314    */
315   public static String checkCache(String url) throws Exception {
316     if (!useCache)
317       return url;
318     if (url.contains("-latest-"))
319       return url;
320     for (String knownUrl : knownUrls) {
321       if (url.startsWith(knownUrl)) {
322         return useCache(url, knownUrl);
323       }
324     }
325     return url;
326   }
327 
328   /**
329    * use the cache for the given URL
330    * 
331    * @param url
332    * @param knownUrl
333    * @return - the URL of the cached file
334    * @throws Exception
335    *           - if the URL is malformed
336    */
337   public static String useCache(String url, String knownUrl) throws Exception {
338     File cacheFile = cacheForUrl(url, knownUrl);
339     if (!cacheFile.exists()) {
340       URL uri = new URL(url);
341       if (debug)
342         LOGGER.log(Level.INFO, "caching to " + cacheFile.getPath());
343       // cache the URL content
344       FileUtils.copyURLToFile(uri, cacheFile);
345     } else {
346       if (debug)
347         LOGGER.log(Level.INFO,
348             "getting cached file from " + cacheFile.getPath());
349     }
350     return cacheFile.toURI().toURL().toExternalForm();
351   }

353   /**
354    * get the cache File for the given url in reference to the given knownUrl
355    * 
356    * @param url
357    * @param knownUrl
358    * @return - the cacheFile
359    */
360   public static File cacheForUrl(String url, String knownUrl) {
361     String filePath = url.substring(knownUrl.length(), url.length());
362     if (cacheRootPath == null)
363       cacheRootPath = System.getProperty("user.home") + java.io.File.separator
364           + ".radolan";
365     File cacheRoot = new File(cacheRootPath);
366     if (!cacheRoot.exists()) {
367       if (debug)
368         LOGGER.log(Level.INFO,
369             "Creating radolan data cache directory " + cacheRoot.getPath());
370       cacheRoot.mkdirs();
371     }
372     File cacheFile = new File(cacheRoot, filePath);
373     return cacheFile;
374   }

405   /**
406    * read all bytes from the given InpuStream - autodetect zipped input
407    * 
408    * @param inputStream
409    * @throws Exception
410    */
411   public void read(InputStream inputStream) throws Exception {
412     bytes = IOUtils.toByteArray(inputStream);
413     // https://tools.ietf.org/html/rfc1952
414     // check for gzip header
415     if (bytes.length < 2) {
416       throw new Exception("input is empty");
417     }
418     int magic = ((bytes[0] & 0xff) << 8) | (bytes[1] & 0xff);
419     if (magic == 0x1f8b) {
420       int zippedLength = bytes.length;
421       InputStream gzStream = new GZIPInputStream(
422           new ByteArrayInputStream(bytes));
423       byte[] unzipped = IOUtils.toByteArray(gzStream);
424       bytes = unzipped;
425       String msg = String.format("unzipped %d to %d bytes", zippedLength,
426           bytes.length);
427       if (debug)
428         LOGGER.log(Level.INFO, msg);
429     }
430     StringBuffer headerBuffer = new StringBuffer();
431     int pos = 0;
432     // read until 0x03 is found or we are way into the binary 2 x typical width
433     // 900 should suffice to terminate ...
434     while (bytes[pos] != 0x03 && pos <= 1800) {
435       headerBuffer.append((char) bytes[pos]);
436       pos++;
437     }
438     headerBuffer.append((char) bytes[pos]);
439     if (pos > 1799 || pos < 21) {
440       throw new Exception("header length " + pos + " out of valid range");
441     }
442     header = headerBuffer.toString();
443     this.parseHeader();
444   }

489   public byte getByte(int x, int y) {
490     int ofs = header.length();

495   /**
496    * get the value at the given x,y coordinate
497    * 
498    * @param x
499    * @param y
500    * @return - the value
501    */
502   public float getValue(int x, int y) {
503     float value = Float.NaN;
504     if (y >= 0 && y < PlainData.length)
505       if (x >= 0 && x < PlainData[y].length)
506         value = PlainData[y][x];
507     return value;
508   }
